{{- /* Repo trend visualization — a lightweight, locally-rendered bar chart for a numeric series (e.g. star history or weekly momentum). Deterministic and dependency-free. Accessible: bars are aria-hidden decoration; the real data is exposed as a visually-hidden description and an optional caption. Params (dict): values : slice of numbers (required) labels : optional slice of strings, aligned to values repo : optional repo slug for the heading/link heading : optional heading text level : optional heading level (default "h3") caption : optional caption / source text */ -}} {{- $values := .values | default slice -}} {{- $labels := .labels | default slice -}} {{- $repo := .repo | default "" -}} {{- /* An empty/omitted repo is treated as absent: fall back to the plain heading rather than emitting a broken " — trend" string. */ -}} {{- $heading := .heading | default (cond (ne $repo "") (printf "%s — trend" $repo) "Trend") -}} {{- $level := .level | default "h3" | lower -}} {{- if not (in (slice "h2" "h3" "h4" "h5" "h6") $level) }}{{ $level = "h3" }}{{ end -}} {{- /* $level is strictly whitelisted above, so the tag name is safe to emit; the heading text itself remains contextually auto-escaped. */ -}} {{- $hOpen := printf `<%s class="repo-trend__heading">` $level | safeHTML -}} {{- $hClose := printf "" $level | safeHTML -}} {{- $caption := .caption -}} {{- if gt (len $values) 0 -}} {{- $max := 0.0 -}} {{- range $values }}{{ if gt (float .) $max }}{{ $max = (float .) }}{{ end }}{{ end -}} {{- if le $max 0.0 }}{{ $max = 1.0 }}{{ end -}} {{- $n := len $values -}} {{- $summary := slice -}} {{- range $i, $v := $values -}} {{- $lab := cond (gt (len $labels) $i) (index $labels $i) (printf "Point %d" (add $i 1)) -}} {{- $summary = $summary | append (printf "%s: %s" $lab (lang.FormatNumber 0 $v)) -}} {{- end -}} {{- end -}}
