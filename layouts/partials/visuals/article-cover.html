{{- /* Article visual orchestrator — guarantees every weekly article gets either a meaningful generated visual or an intentional fallback. Lead approach is locally-generated SVG; a safe local-image path is wired as a documented hook for the image policy (#329). Selection order: 1. visual = "none" -> intentional fallback card 2. compliant LOCAL cover image -> processed, locally-hosted (#329 hook) 3. default -> generated SVG cover card SAFE COVER POLICY (#329): an image is only rendered when `cover.image` resolves to a Hugo resource (page bundle or global asset) — i.e. it is downloaded and locally hosted. External/hotlinked URLs are intentionally ignored here; do NOT rely on fair use or reuse og:image. Attribution + registry handling land in #329. */ -}} {{- $page := . -}} {{- $visual := $page.Params.visual -}} {{- if eq (printf "%v" $visual) "none" -}} {{ partial "visuals/fallback-card.html" (dict "page" $page) }} {{- else -}} {{- $img := "" -}} {{- with $page.Params.cover -}} {{- with .image -}} {{- /* Only accept locally-hosted resources, matched EXACTLY (not as a substring glob) so selection is deterministic and policy-safe. External/hotlinked URLs never resolve to a resource and are ignored. */ -}} {{- /* Escape glob metacharacters (*?[]\) so a cover.image value containing them is matched literally by GetMatch rather than as a glob pattern. */ -}} {{- $pattern := replaceRE `([*?\[\]\\])` `\${1}` . -}} {{- $bundle := $page.Resources.GetMatch $pattern -}} {{- $global := resources.Get . -}} {{- $candidate := or $bundle $global -}} {{- if and $candidate (eq $candidate.ResourceType "image") -}} {{- $img = $candidate -}} {{- end -}} {{- end -}} {{- end -}} {{- if $img -}} {{- $alt := ($page.Params.cover.alt | default $page.Title | plainify) -}} {{- $attribution := $page.Params.cover.attribution -}} {{- else -}} {{ partial "visuals/cover-card.html" (dict "page" $page) }} {{- end -}} {{- end -}}
